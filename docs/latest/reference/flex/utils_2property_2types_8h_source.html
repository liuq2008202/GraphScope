170  int64_t ts_sec = ts / 1000;
171  boost::posix_time::ptime time_point =
172  epoch + boost::posix_time::seconds(ts_sec);
173  boost::posix_time::ptime::date_type date = time_point.date();
174  boost::posix_time::time_duration td = time_point.time_of_day();
